#include <stdio.h>

#define MAX 10

struct Term {
    int coeff; 
    int exp;   
};

int main() {
    struct Term poly1[MAX], poly2[MAX], result[MAX];
    int n1, n2, n3 = 0;
    int i = 0, j = 0;

    printf("Enter number of terms in first polynomial: ");
    scanf("%d", &n1);
    printf("Enter terms (coefficient exponent) in descending order of exponent:\n");
    for (i = 0; i < n1; i++)
        scanf("%d %d", &poly1[i].coeff, &poly1[i].exp);

    printf("Enter number of terms in second polynomial: ");
    scanf("%d", &n2);
    printf("Enter terms (coefficient exponent) in descending order of exponent:\n");
    for (i = 0; i < n2; i++)
        scanf("%d %d", &poly2[i].coeff, &poly2[i].exp);

    i = 0; 
    j = 0;
    while (i < n1 && j < n2) {
        if (poly1[i].exp == poly2[j].exp) {
            result[n3].coeff = poly1[i].coeff + poly2[j].coeff;
            result[n3].exp = poly1[i].exp;
            i++; j++; n3++;
        }
        else if (poly1[i].exp > poly2[j].exp) {
            result[n3].coeff = poly1[i].coeff;
            result[n3].exp = poly1[i].exp;
            i++; n3++;
        }
        else {
            result[n3].coeff = poly2[j].coeff;
            result[n3].exp = poly2[j].exp;
            j++; n3++;
        }
    }

    while (i < n1) {
        result[n3].coeff = poly1[i].coeff;
        result[n3].exp = poly1[i].exp;
        i++; n3++;
    }

    while (j < n2) {
        result[n3].coeff = poly2[j].coeff;
        result[n3].exp = poly2[j].exp;
        j++; n3++;
    }

    printf("\nResultant Polynomial: ");
    for (i = 0; i < n3; i++) {
        if (result[i].coeff != 0) {
            printf("%dX^%d", result[i].coeff, result[i].exp);
            if (i != n3 - 1)
                printf(" + ");
        }
    }
    printf("\n");

    return 0;
}
